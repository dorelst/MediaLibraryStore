            CREATE TABLE user_group (
			`id` int(10) NOT NULL AUTO_INCREMENT,
			`value` varchar(100) NOT NULL,
			PRIMARY KEY(id)
			) ENGINE=INNODB;
			
			CREATE TABLE user (
			`id` int(10) NOT NULL AUTO_INCREMENT,
			`group_id` int(10) NOT NULL,
			`first_name` varchar(50) NOT NULL,
			`last_name` varchar(50) NOT NULL,
			`username` varchar(100) NOT NULL,
			`password` varchar(50) NOT NULL,
			PRIMARY KEY(id),
			CONSTRAINT FK_user_group__user
            FOREIGN KEY (group_id) REFERENCES user_group(id)
			) ENGINE=INNODB;
			
			CREATE TABLE item_type (
			`id` int(10) NOT NULL AUTO_INCREMENT,
			`value` varchar(100) NOT NULL,
			PRIMARY KEY(id)
			) ENGINE=INNODB;
			
			CREATE TABLE media_item (
			`id` int(10) NOT NULL AUTO_INCREMENT,
			`item_type_id` int(10) NOT NULL,
			`title` varchar(100) NOT NULL,
			`author_first_name` varchar(50),
			`author_last_name` varchar(50),
			`released_year` int(10) NOT NULL,
			PRIMARY KEY(id),
			CONSTRAINT FK_item_type__media_item
            FOREIGN KEY (item_type_id) REFERENCES item_type(id)
            ON UPDATE CASCADE
            ON DELETE CASCADE
			) ENGINE=INNODB;
			
			CREATE TABLE booked_item (
            `id` int(10) NOT NULL AUTO_INCREMENT,
            `media_item_id` int(10) NOT NULL,
            `user_id` int(10) NOT NULL,
            `currently_booked` boolean NOT NULL,
            PRIMARY KEY(id),
			CONSTRAINT FK_media_item__booked_item
            FOREIGN KEY (media_item_id) REFERENCES media_item(id)
            ON UPDATE CASCADE
            ON DELETE CASCADE,
			CONSTRAINT FK_user__booked_item
            FOREIGN KEY (user_id) REFERENCES user(id)
            ON UPDATE CASCADE
            ON DELETE CASCADE
            ) ENGINE=INNODB;
			
			CREATE TABLE classroom (
			`id` int(10) NOT NULL AUTO_INCREMENT,
			`classroom_name` varchar(100) NOT NULL,
			PRIMARY KEY(id)
			) ENGINE=INNODB;
			
			CREATE TABLE classroom_member (
			`classroom_id` int(10) NOT NULL,
			`user_id` int(10) NOT NULL,
			PRIMARY KEY(classroom_id, user_id),
			CONSTRAINT FK_user__classroom_member
            FOREIGN KEY (user_id) REFERENCES user(id)
            ON UPDATE CASCADE
            ON DELETE CASCADE,
			CONSTRAINT FK_classroom__classroom_member
            FOREIGN KEY (classroom_id) REFERENCES classroom(id)
            ON UPDATE CASCADE
            ON DELETE CASCADE
			) ENGINE=INNODB;
