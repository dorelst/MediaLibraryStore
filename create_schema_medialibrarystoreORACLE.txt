            CREATE TABLE user_group (
			id NUMBER(10) NOT NULL,
			value VARCHAR2(100) NOT NULL,
			PRIMARY KEY(id)
			);
			
			CREATE TABLE users (
			id NUMBER(10) NOT NULL,
			group_id NUMBER(10) NOT NULL,
			first_name VARCHAR2(50) NOT NULL,
			last_name VARCHAR2(50) NOT NULL,
			username VARCHAR2(100) NOT NULL,
			password VARCHAR2(50) NOT NULL,
			PRIMARY KEY(id),
			CONSTRAINT FK_user_group__user
            			FOREIGN KEY (group_id) REFERENCES user_group(id)
			);
			
			CREATE TABLE item_type (
			id NUMBER(10) NOT NULL,
			value VARCHAR2(100) NOT NULL,
			PRIMARY KEY(id)
			);
			
			CREATE TABLE media_item (
			id NUMBER(10) NOT NULL,
			item_type_id NUMBER(10) NOT NULL,
			title VARCHAR2(100) NOT NULL,
			author_first_name VARCHAR2(50),
			author_last_name VARCHAR2(50),
			released_year NUMBER(10) NOT NULL,
			PRIMARY KEY(id),
			CONSTRAINT FK_item_type__media_item
            			FOREIGN KEY (item_type_id) REFERENCES item_type(id)
            			ON DELETE CASCADE
			);
			
			CREATE TABLE booked_item (
            			id NUMBER(10) NOT NULL,
           			media_item_id NUMBER(10) NOT NULL,
            			user_id NUMBER(10) NOT NULL,
            			currently_booked NUMBER(1) NOT NULL CHECK (currently_booked IN (0,1)),
            			PRIMARY KEY(id),
				CONSTRAINT FK_media_item__booked_item
            				FOREIGN KEY (media_item_id) REFERENCES media_item(id)
            				ON DELETE CASCADE,
				CONSTRAINT FK_user__booked_item
            			FOREIGN KEY (user_id) REFERENCES users(id)
            			ON DELETE CASCADE
            		);
			
			CREATE TABLE classroom (
				id NUMBER(10) NOT NULL,
				classroom_name VARCHAR2(100) NOT NULL,
				PRIMARY KEY(id)
			);
			
			CREATE TABLE classroom_member (
				classroom_id NUMBER(10) NOT NULL,
				user_id NUMBER(10) NOT NULL,
				PRIMARY KEY(classroom_id, user_id),
				CONSTRAINT FK_user__classroom_member
            			FOREIGN KEY (user_id) REFERENCES users(id)
            				ON DELETE CASCADE,
				CONSTRAINT FK_classroom__classroom_member
            			FOREIGN KEY (classroom_id) REFERENCES classroom(id)
            			ON DELETE CASCADE
			);

			CREATE SEQUENCE id_maker
    				MINVALUE 100
    				START WITH 100
    				INCREMENT BY 1;
